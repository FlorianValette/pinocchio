85  inline void framesForwardKinematics(const Model & model,
86  Data & data
87  )
88  {
89  assert(model.check(data) && "data is not consistent with model.");
90 
91  // The following for loop starts by index 1 because the first frame is fixed
92  // and corresponds to the universe.s
93  for (Model::FrameIndex i=1; i < (Model::FrameIndex) model.nframes; ++i)
94  {
95  const Frame & frame = model.frames[i];
96  const Model::JointIndex & parent = frame.parent;
97  if (frame.placement.isIdentity())
98  data.oMf[i] = data.oMi[parent];
99  else
100  data.oMf[i] = data.oMi[parent]*frame.placement;
101  }
102  }

117  template<bool local_frame>
118  inline void getFrameJacobian(const Model & model,
119  const Data & data,
120  const Model::FrameIndex frame_id,
121  Data::Matrix6x & J)
122  {
123  assert(J.cols() == model.nv);
124  assert(data.J.cols() == model.nv);
125  assert(model.check(data) && "data is not consistent with model.");
126 
127  const Model::JointIndex & parent = model.frames[frame_id].parent;
128  const SE3 & oMframe = data.oMf[frame_id];
129  const Frame & frame = model.frames[frame_id];
130 
131  const int colRef = nv(model.joints[parent])+idx_v(model.joints[parent])-1;
132 
133  if(!local_frame)
134  getJacobian<local_frame>(model, data, parent, J);
135 
136  // Lever between the joint center and the frame center expressed in the global frame
137  const SE3::Vector3 lever(data.oMi[parent].rotation() * frame.placement.translation());
138 
139  for(int j=colRef;j>=0;j=data.parents_fromRow[(size_t) j])
140  {
141  if(!local_frame)
142  J.col(j).topRows<3>() -= lever.cross(J.col(j).bottomRows<3>());
143  else
144  J.col(j) = oMframe.actInv(Motion(data.J.col(j))).toVector();
145  }
146  }
