88  {

90  const Scalar N2 = q.squaredNorm();
91 #ifndef NDEBUG
92  const Scalar epsilon = sqrt(sqrt(Eigen::NumTraits<Scalar>::epsilon()));
93  assert(std::fabs(N2-1.) <= epsilon);
94 #endif
95  const Scalar alpha = ((Scalar)3 - N2) / 2;
96  const_cast <Eigen::QuaternionBase<D> &> (q).coeffs() *= alpha;
97 #ifndef NDEBUG
98  const Scalar M = 3 * std::pow((Scalar)1.-epsilon, ((Scalar)-5)/2) / 4;
99  assert(std::fabs(q.norm() - 1) <=
100  std::max(M * sqrt(N2) * (N2 - 1)*(N2 - 1) / 2, Eigen::NumTraits<Scalar>::dummy_precision()));
101 #endif
