74  /* Return the jacobian of the output frame attached to joint <jointId> in the
75  world frame or in the local frame depending on the template argument. The
76  function computeJacobians should have been called first. */
77  template<bool localFrame>
78  void getJacobian(const Model & model,
79  const Data & data,
80  const Model::JointIndex jointId,
81  Data::Matrix6x & J)
82  {
83  assert( J.rows() == data.J.rows() );
84  assert( J.cols() == data.J.cols() );
85  assert(model.check(data) && "data is not consistent with model.");
86 
87  const SE3 & oMjoint = data.oMi[jointId];
88  int colRef = nv(model.joints[jointId])+idx_v(model.joints[jointId])-1;
89  for(int j=colRef;j>=0;j=data.parents_fromRow[(Model::Index)j])
90  {
91  if(! localFrame ) J.col(j) = data.J.col(j);
92  else J.col(j) = oMjoint.actInv(Motion(data.J.col(j))).toVector();
93  }
94  }

143  struct JacobiansTimeVariationForwardStep : public fusion::JointVisitor<JacobiansTimeVariationForwardStep>
144  {
145  typedef boost::fusion::vector <const se3::Model &,
146  se3::Data &,
147  const Eigen::VectorXd &,
148  const Eigen::VectorXd &
149  > ArgsType;
150 
151  JOINT_VISITOR_INIT(JacobiansTimeVariationForwardStep);
152 
153  template<typename JointModel>
154  static void algo(const se3::JointModelBase<JointModel> & jmodel,
155  se3::JointDataBase<typename JointModel::JointDataDerived> & jdata,
156  const se3::Model & model,
157  se3::Data & data,
158  const Eigen::VectorXd & q,
159  const Eigen::VectorXd & v)
160  {
161  const Model::JointIndex & i = (Model::JointIndex) jmodel.id();
162  const Model::JointIndex & parent = model.parents[i];
163 
164  SE3 & oMi = data.oMi[i];
165  Motion & vJ = data.v[i];
166 
167  jmodel.calc(jdata.derived(),q,v);
168 
169  vJ = jdata.v();
170 
171  data.liMi[i] = model.jointPlacements[i]*jdata.M();
172  if(parent>0)
173  {
174  oMi = data.oMi[parent]*data.liMi[i];
175  vJ += data.liMi[i].actInv(data.v[parent]);
176  }
177  else
178  {
179  oMi = data.liMi[i];
180  }
181 
182  jmodel.jointCols(data.J) = oMi.act(jdata.S());
183 
184  // Spatial velocity of joint i expressed in the global frame o
185  const Motion ov(oMi.act(vJ));
186 
187  typedef typename SizeDepType<JointModel::NV>::template ColsReturn<Data::Matrix6x>::Type ColsBlock;
188  ColsBlock dJcols = jmodel.jointCols(data.dJ);
189  ColsBlock Jcols = jmodel.jointCols(data.J);
190 
191  for(int k = 0; k < jmodel.nv(); ++k)
192  dJcols.col(k) = ov.cross(Motion(Jcols.col(k))).toVector();
193  }
194 
195  };
