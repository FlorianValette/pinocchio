50  // Check the validity of the kinematic tree defined by parents.
51  inline bool ParentChecker::checkModel_impl( const Model& model ) const
52  {
53  for( JointIndex j=1;(int)j<model.njoints;++j )
54  if( model.parents[j]>=j ) return false;
55 
56  return true;
57  }

59  inline bool checkData(const Model & model, const Data & data)
60  {
61 #define CHECK_DATA(a) if(!(a)) return false;
62 
63  // TODO JMinvJt,sDUiJt are never explicitly initialized.
64  // TODO impulse_c
65  // They are not check neither
66 
67  CHECK_DATA( (int)data.joints.size() == model.njoints );
68  CHECK_DATA( (int)data.a.size() == model.njoints );
69  CHECK_DATA( (int)data.a_gf.size() == model.njoints );
70  CHECK_DATA( (int)data.v.size() == model.njoints );
71  CHECK_DATA( (int)data.f.size() == model.njoints );
72  CHECK_DATA( (int)data.oMi.size() == model.njoints );
73  CHECK_DATA( (int)data.liMi.size() == model.njoints );
74  CHECK_DATA( (int)data.Ycrb.size() == model.njoints );
75  CHECK_DATA( (int)data.Yaba.size() == model.njoints );
76  CHECK_DATA( (int)data.Fcrb.size() == model.njoints );
77  BOOST_FOREACH(const Data::Matrix6x & F,data.Fcrb) CHECK_DATA( F.cols() == model.nv );
78  CHECK_DATA( (int)data.iMf.size() == model.njoints );
79  CHECK_DATA( (int)data.iMf.size() == model.njoints );
80  CHECK_DATA( (int)data.com.size() == model.njoints );
81  CHECK_DATA( (int)data.vcom.size() == model.njoints );
82  CHECK_DATA( (int)data.acom.size() == model.njoints );
83  CHECK_DATA( (int)data.mass.size() == model.njoints );
84 
85  CHECK_DATA( data.tau.size() == model.nv );
86  CHECK_DATA( data.nle.size() == model.nv );
87  CHECK_DATA( data.ddq.size() == model.nv );
88  CHECK_DATA( data.u.size() == model.nv );
89  CHECK_DATA( data.M.rows() == model.nv );
90  CHECK_DATA( data.M.cols() == model.nv );
91  CHECK_DATA( data.Ag.cols() == model.nv );
92  CHECK_DATA( data.U.cols() == model.nv );
93  CHECK_DATA( data.U.rows() == model.nv );
94  CHECK_DATA( data.D.size() == model.nv );
95  CHECK_DATA( data.tmp.size() == model.nv );
96  CHECK_DATA( data.J.cols() == model.nv );
97  CHECK_DATA( data.Jcom.cols() == model.nv );
98  CHECK_DATA( data.torque_residual.size() == model.nv );
99  CHECK_DATA( data.dq_after.size() == model.nv );
100  //CHECK_DATA( data.impulse_c.size()== model.nv );
101 
102  CHECK_DATA( (int)data.oMf.size() == model.nframes );
103 
104  CHECK_DATA( (int)data.lastChild.size() == model.njoints );
105  CHECK_DATA( (int)data.nvSubtree.size() == model.njoints );
106  CHECK_DATA( (int)data.parents_fromRow.size() == model.nv );
107  CHECK_DATA( (int)data.nvSubtree_fromRow.size() == model.nv );
108 
109  for( JointIndex j=1;int(j)<model.njoints;++j )
110  {
111  JointIndex c = (JointIndex)data.lastChild[j];
112  CHECK_DATA((int)c<model.njoints);
113  int nv=model.joints[j].nv();
114  for( JointIndex d=j+1;d<=c;++d ) // explore all descendant
115  {
116  CHECK_DATA( model.parents[d]>=j );
117  nv+=model.joints[d].nv();
118  }
119  CHECK_DATA(nv==data.nvSubtree[j]);
120 
121  for( JointIndex d=c+1;(int)d<model.njoints;++d)
122  CHECK_DATA( (model.parents[d]<j)||(model.parents[d]>c) );
123 
124  int row = model.joints[j].idx_v();
125  CHECK_DATA(data.nvSubtree[j] == data.nvSubtree_fromRow[(size_t)row]);
126 
127  const JointModel & jparent = model.joints[model.parents[j]];
128  if(row==0) { CHECK_DATA(data.parents_fromRow[(size_t)row]==-1); }
129  else { CHECK_DATA(jparent.idx_v()+jparent.nv()-1 == data.parents_fromRow[(size_t)row]); }
130  }
131 
132 #undef CHECK_DATA
133  return true;
134  }
