220  inline Motion operator^( const Motion& m1, const MotionRevolute<0>& m2)
221  {
222  /* nu1^nu2 = ( v1^w2+w1^v2, w1^w2 )
223  * nu1^(0,w2) = ( v1^w2 , w1^w2 )
224  * (x,y,z)^(w,0,0) = ( 0,zw,-yw )
225  * nu1^(0,wx) = ( 0,vz1 wx,-vy1 wx, 0,wz1 wx,-wy1 wx)
226  */
227  const Motion::Vector3& v = m1.linear();
228  const Motion::Vector3& w = m1.angular();
229  const double & wx = m2.w;
230  return Motion( Motion::Vector3(0,v[2]*wx,-v[1]*wx),
231  Motion::Vector3(0,w[2]*wx,-w[1]*wx)
232  );
233  }
234 
235  inline Motion operator^( const Motion& m1, const MotionRevolute<1>& m2)
236  {
237  /* nu1^nu2 = ( v1^w2+w1^v2, w1^w2 )
238  * nu1^(0,w2) = ( v1^w2 , w1^w2 )
239  * (x,y,z)^(0,w,0) = ( -z,0,x )
240  * nu1^(0,wx) = ( -vz1 wx,0,vx1 wx, -wz1 wx,0,wx1 wx)
241  */
242  const Motion::Vector3& v = m1.linear();
243  const Motion::Vector3& w = m1.angular();
244  const double & wx = m2.w;
245  return Motion( Motion::Vector3(-v[2]*wx,0, v[0]*wx),
246  Motion::Vector3(-w[2]*wx,0, w[0]*wx)
247  );
248  }

250  inline Motion operator^( const Motion& m1, const MotionRevolute<2>& m2)
251  {
252  /* nu1^nu2 = ( v1^w2+w1^v2, w1^w2 )
253  * nu1^(0,w2) = ( v1^w2 , w1^w2 )
254  * (x,y,z)^(0,0,w) = ( y,-x,0 )
255  * nu1^(0,wx) = ( vy1 wx,-vx1 wx,0, wy1 wx,-wx1 wx,0 )
256  */
257  const Motion::Vector3& v = m1.linear();

261  Motion::Vector3(w[1]*wx,-w[0]*wx,0)
262  );
