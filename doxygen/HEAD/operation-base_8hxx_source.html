207  template <class Derived>
208  template <class ConfigL_t, class ConfigR_t, class ConfigOut_t>
209  void LieGroupOperationBase<Derived>::interpolate_impl(
210  const Eigen::MatrixBase<ConfigL_t> & q0,
211  const Eigen::MatrixBase<ConfigR_t> & q1,
212  const Scalar& u,
213  const Eigen::MatrixBase<ConfigOut_t>& qout)
214  {
215  if (u == 0) const_cast<Eigen::MatrixBase<ConfigOut_t>&>(qout) = q0;
216  else if(u == 1) const_cast<Eigen::MatrixBase<ConfigOut_t>&>(qout) = q1;
217  else integrate(q0, u * difference(q0, q1), qout);
218  }
