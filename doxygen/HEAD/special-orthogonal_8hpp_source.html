92  template <class ConfigIn_t, class Velocity_t, class ConfigOut_t>
93  static void integrate_impl(const Eigen::MatrixBase<ConfigIn_t> & q,
94  const Eigen::MatrixBase<Velocity_t> & v,
95  const Eigen::MatrixBase<ConfigOut_t> & qout)
96  {
97  ConfigOut_t& out = (const_cast< Eigen::MatrixBase<ConfigOut_t>& >(qout)).derived();
98 
99  const Scalar & ca = q(0);
100  const Scalar & sa = q(1);
101  const Scalar & omega = v(0);
102 
103  Scalar cosOmega,sinOmega; SINCOS(omega, &sinOmega, &cosOmega);
104  // TODO check the cost of atan2 vs SINCOS
105 
106  out << cosOmega * ca - sinOmega * sa,
107  sinOmega * ca + cosOmega * sa;
108  const Scalar norm2 = q.squaredNorm();
109  out *= (3 - norm2) / 2;
110  }

112  template <class ConfigL_t, class ConfigR_t, class ConfigOut_t>
113  static void interpolate_impl(const Eigen::MatrixBase<ConfigL_t> & q0,
114  const Eigen::MatrixBase<ConfigR_t> & q1,
115  const Scalar& u,
116  const Eigen::MatrixBase<ConfigOut_t>& qout)
117  {
118  ConfigOut_t& out = (const_cast< Eigen::MatrixBase<ConfigOut_t>& >(qout)).derived();
119 
120  assert ( (q0.norm() - 1) < 1e-8 && "initial configuration not normalized");
121  assert ( (q1.norm() - 1) < 1e-8 && "final configuration not normalized");
122  Scalar cosTheta = q0.dot(q1);
123  Scalar sinTheta = q0(0)*q1(1) - q0(1)*q1(0);
124  Scalar theta = atan2(sinTheta, cosTheta);
125  assert (fabs (sin (theta) - sinTheta) < 1e-8);
126 
127  if (fabs (theta) > 1e-6 && fabs (theta) < PI - 1e-6)
128  {
129  out = (sin ((1-u)*theta)/sinTheta) * q0
130  + (sin ( u *theta)/sinTheta) * q1;
131  }
132  else if (fabs (theta) < 1e-6) // theta = 0
133  {
134  out = (1-u) * q0 + u * q1;
135  }
136  else // theta = +-PI
137  {
138  double theta0 = atan2 (q0(1), q0(0));
139  out << cos (theta0 + u * theta),
140  sin (theta0 + u * theta);
141  }
142  }

203  const Eigen::MatrixBase<ConfigR_t> & q1,

207  ConstQuaternionMap_t p1 (q1.derived().data());
