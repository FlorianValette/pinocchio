140  ConfigVector_t integrate_impl(const Eigen::VectorXd & qs,const Eigen::VectorXd & vs) const
141  {
142  typename Eigen::VectorXd::ConstFixedSegmentReturnType<NQ>::Type & q = qs.segment<NQ> (idx_q ());
143  typename Eigen::VectorXd::ConstFixedSegmentReturnType<NV>::Type & q_dot = vs.segment<NV> (idx_v ());
144 
145  const double & ca = q(0);
146  const double & sa = q(1);
147  const double & omega = q_dot(0);
148 
149  double cosOmega,sinOmega; SINCOS(omega, &sinOmega, &cosOmega);
150  // TODO check the cost of atan2 vs SINCOS
151 
152  ConfigVector_t result (cosOmega * ca - sinOmega * sa,
153  sinOmega * ca + cosOmega * sa);
154  const double norm2 = q.squaredNorm();
155  result *= (3 - norm2) / 2;
156 
157  return result;
158  }

161  ConfigVector_t interpolate_impl(const Eigen::VectorXd & q0,const Eigen::VectorXd & q1, const double u) const
162  {
163  typename Eigen::VectorXd::ConstFixedSegmentReturnType<NQ>::Type & qi = q0.segment<NQ> (idx_q ());
164  typename Eigen::VectorXd::ConstFixedSegmentReturnType<NQ>::Type & qf = q1.segment<NQ> (idx_q ());
165 
166  const double & c0 = qi(0), s0 = qi(1);
167  const double & c1 = qf(0), s1 = qf(1);
168 
169  assert ( (qi.norm() - 1) < 1e-8 && "initial configuration not normalized");
170  assert ( (qf.norm() - 1) < 1e-8 && "final configuration not normalized");
171  double cosTheta = c0*c1 + s0*s1;
172  double sinTheta = c0*s1 - s0*c1;
173  double theta = atan2(sinTheta, cosTheta);
174  assert (fabs (sin (theta) - sinTheta) < 1e-8);
175 
176  ConfigVector_t result;
177  if (fabs (theta) > 1e-6 && fabs (theta) < PI - 1e-6)
178  {
179  result = (sin ((1-u)*theta)/sinTheta) * qi
180  + (sin (u*theta)/sinTheta) * qf;
181  }
182  else if (fabs (theta) < 1e-6) // theta = 0
183  {
184  result = (1-u) * qi + u * qf;
185  }
186  else // theta = +-PI
187  {
188  double theta0 = atan2 (s0, c0);
189  result << cos (theta0 + u * theta),
190  sin (theta0 + u * theta);
191  }
192 
193  return result;
194  }

215  TangentVector_t difference_impl(const Eigen::VectorXd & q0,const Eigen::VectorXd & q1) const
216  {
217  typename Eigen::VectorXd::ConstFixedSegmentReturnType<NQ>::Type & qi = q0.segment<NQ> (idx_q ());
218  typename Eigen::VectorXd::ConstFixedSegmentReturnType<NQ>::Type & qf = q1.segment<NQ> (idx_q ());
219 
220  const double & c0 = qi(0), s0 = qi(1);
221  const double & c1 = qf(0), s1 = qf(1);
222 
223  TangentVector_t result;
224  result << atan2 (s1*c0 - s0*c1, c0*c1 + s0*s1);
225  return result;
226  }
