38  inline Inertia convertFromUrdf (const ::urdf::Inertial & Y)
39  {
40  const ::urdf::Vector3 & p = Y.origin.position;
41  const ::urdf::Rotation & q = Y.origin.rotation;
42 
43  const Eigen::Vector3d com(p.x,p.y,p.z);
44  const Eigen::Matrix3d & R = Eigen::Quaterniond(q.w,q.x,q.y,q.z).matrix();
45 
46  Eigen::Matrix3d I; I <<
47  Y.ixx,Y.ixy,Y.ixz,
48  Y.ixy,Y.iyy,Y.iyz,
49  Y.ixz,Y.iyz,Y.izz;
50  return Inertia(Y.mass,com,R*I*R.transpose());
51  }
