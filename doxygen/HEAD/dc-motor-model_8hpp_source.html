95  void ode_func(ActuatorDCMotorData<Scalar_> &data)
96  {
97  data.dstate[0] = (data.rotorResistor * data.state[0] + data.V - data.backEMF * data.state[2])/
98  data.terminalInductance;
99  data.dstate[1] = data.state[2];
100  data.dstate[2] = (data.state[0] * data.torqueConst - data.tau_l /data.gearRatio - data.speedTorqueGrad * data.gearRatio * data.state[2])/(data.rotorInertia * data.gearRatio);
101  data.motor_torque = data.state[0] * data.torqueConst;
102  }
