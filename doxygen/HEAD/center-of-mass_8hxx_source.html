63  template<bool do_position, bool do_velocity, bool do_acceleration>
64  inline void centerOfMass(const Model & model, Data & data,
65  const bool computeSubtreeComs)
66  {
67  assert(model.check(data) && "data is not consistent with model.");
68  using namespace se3;
69 
70  data.mass[0] = 0;
71  if(do_position)
72  data.com[0].setZero ();
73  if(do_velocity)
74  data.vcom[0].setZero ();
75  if(do_acceleration)
76  data.acom[0].setZero ();
77 
78  // Forward Step
79  for(Model::JointIndex i=1;i<(Model::JointIndex)(model.njoints);++i)
80  {
81  const double mass = model.inertias[i].mass();
82  const SE3::Vector3 & lever = model.inertias[i].lever();
83 
84  const Motion & v = data.v[i];
85  const Motion & a = data.a[i];
86 
87  data.mass[i] = mass;
88 
89  if(do_position)
90  data.com[i] = mass * lever;
91 
92  if(do_velocity)
93  data.vcom[i] = mass * (v.angular().cross(lever) + v.linear());
94 
95  if(do_acceleration)
96  data.acom[i] = mass * (a.angular().cross(lever) + a.linear())
97  + v.angular().cross(data.vcom[i]); // take into accound the coriolis part of the acceleration
98  }
99 
100  // Backward Step
101  for(Model::JointIndex i=(Model::JointIndex)(model.njoints-1); i>0; --i)
102  {
103  const Model::JointIndex & parent = model.parents[i];
104  const SE3 & liMi = data.liMi[i];
105 
106  data.mass[parent] += data.mass[i];
107 
108  if(do_position)
109  data.com[parent] += (liMi.rotation()*data.com[i]
110  + data.mass[i] * liMi.translation());
111 
112  if(do_velocity)
113  data.vcom[parent] += liMi.rotation()*data.vcom[i];
114  data.acom[parent] += liMi.rotation()*data.acom[i];
115 
116  if(computeSubtreeComs)
117  {
118  if(do_position)
119  data.com[i] /= data.mass[i];
120  if(do_velocity)
121  data.vcom[i] /= data.mass[i];
122  if(do_acceleration)
123  data.acom[i] /= data.mass[i];
124  }
125  }
126 
127  if(do_position)
128  data.com[0] /= data.mass[0];
129  if(do_velocity)
130  data.vcom[0] /= data.mass[0];
131  if(do_acceleration)
132  data.acom[0] /= data.mass[0];
133  }

146  struct JacobianCenterOfMassBackwardStep
147  : public fusion::JointVisitor<JacobianCenterOfMassBackwardStep>
148  {
149  typedef boost::fusion::vector<const se3::Model &,
150  se3::Data &,
151  const bool
152  > ArgsType;
153 
154  JOINT_VISITOR_INIT(JacobianCenterOfMassBackwardStep);

156  template<typename JointModel>
157  static void algo(const se3::JointModelBase<JointModel> & jmodel,
158  se3::JointDataBase<typename JointModel::JointDataDerived> & jdata,
159  const se3::Model& model,
160  se3::Data& data,
161  const bool computeSubtreeComs )
162  {
163  const Model::JointIndex & i = (Model::JointIndex) jmodel.id();
164  const Model::JointIndex & parent = model.parents[i];
165 
166  data.com[parent] += data.com[i];
167  data.mass[parent] += data.mass[i];
168 
169  typedef Data::Matrix6x Matrix6x;
170  typedef typename SizeDepType<JointModel::NV>::template ColsReturn<Matrix6x>::Type ColBlock;
171 
172  ColBlock Jcols = jmodel.jointCols(data.J);
173  Jcols = data.oMi[i].act(jdata.S());
174 
175  if (JointModel::NV == -1)
176  {
177  if( jmodel.nv() ==1 )
178  {
179  data.Jcom.col(jmodel.idx_v())
180  = data.mass[i] * Jcols.template topLeftCorner<3,1>()
181  - data.com[i].cross(Jcols.template bottomLeftCorner<3,1>()) ;
182  }
183  else
184  {
185  jmodel.jointCols(data.Jcom)
186  = data.mass[i] * Jcols.template topRows<3>()
187  - skew(data.com[i]) * Jcols.template bottomRows<3>();
188  }
189  }
190  else
191  {
192  if( JointModel::NV ==1 )
193  {
194  data.Jcom.col(jmodel.idx_v())
195  = data.mass[i] * Jcols.template topLeftCorner<3,1>()
196  - data.com[i].cross(Jcols.template bottomLeftCorner<3,1>()) ;
197  }
198  else
199  {
200  jmodel.jointCols(data.Jcom)
201  = data.mass[i] * Jcols.template topRows<3>()
202  - skew(data.com[i]) * Jcols.template bottomRows<3>();
203  }
204  }
205 
206 
207  if(computeSubtreeComs)
208  data.com[i] /= data.mass[i];
209  }
210 
211  };

213  inline const Data::Matrix3x &
214  jacobianCenterOfMass(const Model & model, Data & data,
215  const Eigen::VectorXd & q,
216  const bool computeSubtreeComs,
217  const bool updateKinematics)
218  {
219  assert(model.check(data) && "data is not consistent with model.");
220  data.com[0].setZero ();
221  data.mass[0] = 0;
222 
223  // Forward step
224  if (updateKinematics)
225  forwardKinematics(model, data, q);
226 
227  for(Model::JointIndex i=1;i<(Model::JointIndex)(model.njoints);++i)
228  {
229  const double mass = model.inertias[i].mass();
230  const SE3::Vector3 & lever = model.inertias[i].lever();
231 
232  data.mass[i] = mass;
233  data.com[i] = mass*data.oMi[i].act(lever);
234  }
235 
236  // Backward step
237  for( Model::JointIndex i= (Model::JointIndex) (model.njoints-1);i>0;--i )
238  {
239  JacobianCenterOfMassBackwardStep
240  ::run(model.joints[i],data.joints[i],
241  JacobianCenterOfMassBackwardStep::ArgsType(model,data,computeSubtreeComs));
242  }
243 
244  data.com[0] /= data.mass[0];
245  data.Jcom /= data.mass[0];
246 
247  return data.Jcom;
248  }

250  inline const Data::Matrix3x &
251  getJacobianComFromCrba(const Model & model, Data & data)
252  {
253  assert(model.check(data) && "data is not consistent with model.");
254  const SE3 & oM1 = data.liMi[1];
255 
256  // As the 6 first rows of M*a are a wrench, we just need to multiply by the
257  // relative rotation between the first joint and the world
258  const SE3::Matrix3 oR1_over_m (oM1.rotation() / data.M(0,0));
259 
260  // I don't know why, but the colwise multiplication is much more faster
261  // than the direct Eigen multiplication
262  for (long k=0; k<model.nv;++k)
263  data.Jcom.col(k) = oR1_over_m * data.M.topRows<3> ().col(k);
264  return data.Jcom;
265  }
