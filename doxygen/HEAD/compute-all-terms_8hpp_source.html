70  template<typename JointModel>
71  static void algo(const se3::JointModelBase<JointModel> & jmodel,
72  se3::JointDataBase<typename JointModel::JointDataDerived> & jdata,
73  const se3::Model & model,
74  se3::Data & data,
75  const Eigen::VectorXd & q,
76  const Eigen::VectorXd & v)
77  {
78  using namespace Eigen;
79  using namespace se3;
80 
81  const Model::JointIndex & i = (Model::JointIndex) jmodel.id();
82  const Model::JointIndex & parent = model.parents[i];
83 
84  jmodel.calc(jdata.derived(),q,v);
85 
86  // CRBA
87  data.liMi[i] = model.jointPlacements[i]*jdata.M();
88  data.Ycrb[i] = model.inertias[i];
89 
90  // Jacobian + NLE
91  data.v[i] = jdata.v();
92 
93  if(parent>0)
94  {
95  data.oMi[i] = data.oMi[parent]*data.liMi[i];
96  data.v[i] += data.liMi[i].actInv(data.v[parent]);
97  }
98  else
99  data.oMi[i] = data.liMi[i];
100 
101  jmodel.jointCols(data.J) = data.oMi[i].act(jdata.S());
102 
103  data.a_gf[i] = data.a[i] = jdata.c() + (data.v[i] ^ jdata.v());
104  if (parent > 0)
105  data.a[i] += data.liMi[i].actInv(data.a[parent]);
106 
107  data.a_gf[i] += data.liMi[i].actInv(data.a_gf[parent]);
108 
109  data.f[i] = model.inertias[i]*data.a_gf[i] + model.inertias[i].vxiv(data.v[i]); // -f_ext
110 
111  // CoM
112  const double mass = model.inertias[i].mass();
113  const SE3::Vector3 & lever = model.inertias[i].lever();
114 
115  data.com[i] = mass * lever;
116  data.mass[i] = mass;
117 
118  data.vcom[i] = mass * (data.v[i].angular().cross(lever) + data.v[i].linear());
119  }
120 
121  };

130  template<typename JointModel>
131  static void algo(const JointModelBase<JointModel> & jmodel,
132  JointDataBase<typename JointModel::JointDataDerived> & jdata,
133  const Model & model,
134  Data & data)
135  {
136  /*
137  * F[1:6,i] = Y*S
138  * M[i,SUBTREE] = S'*F[1:6,SUBTREE]
139  * if li>0
140  * Yli += liXi Yi
141  * F[1:6,SUBTREE] = liXi F[1:6,SUBTREE]
142  */
143  const Model::JointIndex & i = (Model::JointIndex) jmodel.id();
144  const Model::JointIndex & parent = model.parents[i];
145  const SE3 & oMi = data.oMi[i];
146 
147  /* F[1:6,i] = Y*S */
148  jmodel.jointCols(data.Fcrb[i]) = data.Ycrb[i] * jdata.S();
149 
150  /* M[i,SUBTREE] = S'*F[1:6,SUBTREE] */
151  data.M.block(jmodel.idx_v(),jmodel.idx_v(),jmodel.nv(),data.nvSubtree[i])
152  = jdata.S().transpose()*data.Fcrb[i].middleCols(jmodel.idx_v(),data.nvSubtree[i]);
153 
154 
155  jmodel.jointVelocitySelector(data.nle) = jdata.S().transpose()*data.f[i];
156  if(parent>0)
157  {
158  /* Yli += liXi Yi */
159  data.Ycrb[parent] += data.liMi[i].act(data.Ycrb[i]);
160 
161  /* F[1:6,SUBTREE] = liXi F[1:6,SUBTREE] */
162  Eigen::Block<typename Data::Matrix6x> jF
163  = data.Fcrb[parent].block(0,jmodel.idx_v(),6,data.nvSubtree[i]);
164  Eigen::Block<typename Data::Matrix6x> iF
165  = data.Fcrb[i].block(0,jmodel.idx_v(),6,data.nvSubtree[i]);
166  forceSet::se3Action(data.liMi[i], iF, jF);
167 
168  data.f[parent] += data.liMi[i].act(data.f[i]);
169  }
170 
171  // CoM
172  const SE3 & liMi = data.liMi[i];
173 
174  data.com[parent] += (liMi.rotation()*data.com[i]
175  + data.mass[i] * liMi.translation());
176 
177  SE3::Vector3 com_in_world (oMi.rotation() * data.com[i] + data.mass[i] * oMi.translation());
178 
179  data.vcom[parent] += liMi.rotation()*data.vcom[i];
180  data.mass[parent] += data.mass[i];
181 
182  typedef Data::Matrix6x Matrix6x;
183  typedef typename SizeDepType<JointModel::NV>::template ColsReturn<Matrix6x>::Type ColBlock;
184 
185  ColBlock Jcols = jmodel.jointCols(data.J);
186 
187  if( JointModel::NV==1 )
188  data.Jcom.col(jmodel.idx_v())
189  = data.mass[i] * Jcols.template topLeftCorner<3,1>()
190  - com_in_world.cross(Jcols.template bottomLeftCorner<3,1>()) ;
191  else
192  jmodel.jointCols(data.Jcom)
193  = data.mass[i] * Jcols.template topRows<3>()
194  - skew(com_in_world) * Jcols.template bottomRows<3>();
195 
196  data.com[i] /= data.mass[i];
197  data.vcom[i] /= data.mass[i];
198  }
199  };

201  inline void
202  computeAllTerms(const Model & model,
203  Data & data,
204  const Eigen::VectorXd & q,
205  const Eigen::VectorXd & v)
206  {
207  assert(model.check(data) && "data is not consistent with model.");
208  data.v[0].setZero();
209  data.a[0].setZero();
210  data.a_gf[0] = -model.gravity;
211 
212  data.mass[0] = 0;
213  data.com[0].setZero ();
214  data.vcom[0].setZero ();
215 
216  for(Model::JointIndex i=1;i<(Model::JointIndex) model.njoints;++i)
217  {
218  CATForwardStep::run(model.joints[i],data.joints[i],
219  CATForwardStep::ArgsType(model,data,q,v));
220  }
221 
222  for(Model::JointIndex i=(Model::JointIndex)(model.njoints-1);i>0;--i)
223  {
224  CATBackwardStep::run(model.joints[i],data.joints[i],
225  CATBackwardStep::ArgsType(model,data));
226  }
227 
228  // CoM
229  data.com[0] /= data.mass[0];
230  data.vcom[0] /= data.mass[0];
231 
232  // JCoM
233  data.Jcom /= data.mass[0];
234 
235  // Energy
236  kineticEnergy(model, data, q, v, false);
237  potentialEnergy(model, data, q, false);
238 
239  }
