89  template <class ConfigL_t, class ConfigR_t, class Tangent_t>
90  static void difference_impl(const Eigen::MatrixBase<ConfigL_t> & q0,
91  const Eigen::MatrixBase<ConfigR_t> & q1,
92  const Eigen::MatrixBase<Tangent_t> & d)
93  {
94  SE3 M0(SE3::Identity()); forwardKinematics(M0, q0);
95  SE3 M1(SE3::Identity()); forwardKinematics(M1, q1);
96 
97  Motion nu(log6(M0.inverse()*M1)); // TODO: optimize implementation
98 
99  Tangent_t& out = const_cast< Eigen::MatrixBase<Tangent_t>& >(d).derived();
100  out.template head<2>() = nu.linear().head<2>();
101  out(2) = nu.angular()(2);
102  }

104  template <class ConfigIn_t, class Velocity_t, class ConfigOut_t>
105  static void integrate_impl(const Eigen::MatrixBase<ConfigIn_t> & q,
106  const Eigen::MatrixBase<Velocity_t> & vs,
107  const Eigen::MatrixBase<ConfigOut_t> & qout)
108  {
109  ConfigOut_t& out = (const_cast< Eigen::MatrixBase<ConfigOut_t>& >(qout)).derived();
110  typedef Eigen::Matrix<Scalar, 2, 2> Matrix22;
111  typedef Eigen::Matrix<Scalar, 2, 1> Vector2;
112 
113  const double& c0 = q(2), s0 = q(3);
114  Matrix22 R0;
115  R0 << c0, -s0, s0, c0;
116 
117  const double& t = vs[2];
118  const double theta = std::fabs(t);
119 
120  if(theta > 1e-14)
121  {
122  // vs = [ x, y, t ]
123  // w = [ 0, 0, t ]
124  // v = [ x, y, 0 ]
125  // Considering only the 2x2 top left corner:
126  // Sp = [ 0, -1; 1, 0],
127  // if t > 0: S = Sp
128  // else : S = -Sp
129  // S / t = Sp / |t|
130  // S * S = - I2
131  // R = I2 + ( 1 - ct) / |t| * S + ( 1 - st / |t| ) * S * S
132  // = ( 1 - ct) / |t| * S + st / |t| * I2
133  //
134  // Ru = exp3 (w)
135  // tu = R * v = (1 - ct) / |t| * S * v + st / t * v
136  //
137  // M0 * Mu = ( R0 * Ru, R0 * tu + t0 )
138 
139  // FIXME remove this copy.
140  Vector2 v = vs.template head<2>();
141  Eigen::Matrix<Scalar,2,1> cst;
142  SINCOS (t, &cst[1], &cst[0]);
143  const Scalar inv_theta = 1/theta;
144  const Scalar c_coeff = (1.-cst[0]) * inv_theta;
145  const Scalar s_coeff = std::fabs(cst[1]) * inv_theta;
146  const Vector2 Sp_v (-v[1], v[0]);
147 
148  if (t > 0) out.template head<2>() = q.template head<2>() + R0 * (s_coeff * v + c_coeff * Sp_v);
149  else out.template head<2>() = q.template head<2>() + R0 * (s_coeff * v - c_coeff * Sp_v);
150  out.template tail<2>() = R0 * cst;
151  }
152  else
153  {
154  // cos(t) ~ 1 - t^2 / 2
155  // sin(t) ~ t
156  out.template head<2>() = q.template head<2>() + R0*vs.template head<2>();
157  out(2) = c0 * 1 - s0 * t;
158  out(3) = s0 * 1 + c0 * t;
159  }
160  }

248  template <class ConfigL_t, class ConfigR_t, class Tangent_t>
249  static void difference_impl(const Eigen::MatrixBase<ConfigL_t> & q0,
250  const Eigen::MatrixBase<ConfigR_t> & q1,
251  const Eigen::MatrixBase<Tangent_t> & d)
252  {
253  ConstQuaternionMap_t p0 (q0.derived().template tail<4>().data());
254  ConstQuaternionMap_t p1 (q1.derived().template tail<4>().data());
255  const_cast < Eigen::MatrixBase<Tangent_t>& > (d)
256  = log6( SE3(p0.matrix(), q0.derived().template head<3>()).inverse()
257  * SE3(p1.matrix(), q1.derived().template head<3>())).toVector();
258  }

260  template <class ConfigIn_t, class Velocity_t, class ConfigOut_t>
261  static void integrate_impl(const Eigen::MatrixBase<ConfigIn_t> & q,
262  const Eigen::MatrixBase<Velocity_t> & v,
263  const Eigen::MatrixBase<ConfigOut_t> & qout)
264  {
265  ConfigOut_t& out = (const_cast< Eigen::MatrixBase<ConfigOut_t>& >(qout)).derived();
266  ConstQuaternionMap_t quat(q.derived().template tail<4>().data());
267  QuaternionMap_t res_quat (out.template tail<4>().data());
268 
269  SE3 M0 (quat.matrix(), q.derived().template head<3>());
270  SE3 M1 (M0 * exp6(Motion(v)));
271 
272  out.template head<3>() = M1.translation();
273  res_quat = M1.rotation();
274  // Norm of qs might be epsilon-different to 1, so M1.rotation might be epsilon-different to a rotation matrix.
275  // It is then safer to re-normalized after converting M1.rotation to quaternion.
276  firstOrderNormalize(res_quat);
277  }
