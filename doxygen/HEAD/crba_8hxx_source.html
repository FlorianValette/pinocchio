62  template<typename JointModel>
63  static void algo(const JointModelBase<JointModel> & jmodel,
64  JointDataBase<typename JointModel::JointDataDerived> & jdata,
65  const Model & model,
66  Data & data)
67  {
68  /*
69  * F[1:6,i] = Y*S
70  * M[i,SUBTREE] = S'*F[1:6,SUBTREE]
71  * if li>0
72  * Yli += liXi Yi
73  * F[1:6,SUBTREE] = liXi F[1:6,SUBTREE]
74  */
75  typedef Data::Matrix6x::ColsBlockXpr Block;
76  const Model::JointIndex & i = (Model::JointIndex) jmodel.id();
77 
78  /* F[1:6,i] = Y*S */
79  //data.Fcrb[i].block<6,JointModel::NV>(0,jmodel.idx_v()) = data.Ycrb[i] * jdata.S();
80  jmodel.jointCols(data.Fcrb[i]) = data.Ycrb[i] * jdata.S();
81 
82  /* M[i,SUBTREE] = S'*F[1:6,SUBTREE] */
83  data.M.block(jmodel.idx_v(),jmodel.idx_v(),jmodel.nv(),data.nvSubtree[i])
84  = jdata.S().transpose()*data.Fcrb[i].middleCols(jmodel.idx_v(),data.nvSubtree[i]);
85 
86  const Model::JointIndex & parent = model.parents[i];
87  if(parent>0)
88  {
89  /* Yli += liXi Yi */
90  data.Ycrb[parent] += data.liMi[i].act(data.Ycrb[i]);
91 
92  /* F[1:6,SUBTREE] = liXi F[1:6,SUBTREE] */
93  Block jF
94  = data.Fcrb[parent].middleCols(jmodel.idx_v(),data.nvSubtree[i]);
95  Block iF
96  = data.Fcrb[i].middleCols(jmodel.idx_v(),data.nvSubtree[i]);
97  forceSet::se3Action(data.liMi[i], iF, jF);
98  }
99  }

102  inline const Eigen::MatrixXd&
103  crba(const Model & model, Data& data,
104  const Eigen::VectorXd & q)
105  {
106  assert(model.check(data) && "data is not consistent with model.");
107 
108  for( Model::JointIndex i=1;i<(Model::JointIndex)(model.njoints);++i )
109  {
110  CrbaForwardStep::run(model.joints[i],data.joints[i],
111  CrbaForwardStep::ArgsType(model,data,q));
112  }
113 
114  for( Model::JointIndex i=(Model::JointIndex)(model.njoints-1);i>0;--i )
115  {
116  CrbaBackwardStep::run(model.joints[i],data.joints[i],
117  CrbaBackwardStep::ArgsType(model,data));
118  }
119 
120  return data.M;

204  inline const Data::Matrix6x &
205  ccrba(const Model & model, Data & data,
206  const Eigen::VectorXd & q,
207  const Eigen::VectorXd & v)
208  {
209  assert(model.check(data) && "data is not consistent with model.");
210  typedef Eigen::Block <Data::Matrix6x,3,-1> Block3x;
211 
212  forwardKinematics(model, data, q);
213  data.Ycrb[0].setZero();
214  for(Model::Index i=1;i<(Model::Index)(model.njoints);++i )
215  data.Ycrb[i] = model.inertias[i];
216 
217 
218  for(Model::Index i=(Model::Index)(model.njoints-1);i>0;--i)
219  {
220  CcrbaBackwardStep::run(model.joints[i],data.joints[i],
221  CcrbaBackwardStep::ArgsType(model,data));
222  }
223  data.com[0] = data.Ycrb[0].lever();
224 
225  const Block3x Ag_lin = data.Ag.middleRows<3> (Force::LINEAR);
226  Block3x Ag_ang = data.Ag.middleRows<3> (Force::ANGULAR);
227  for (long i = 0; i<model.nv; ++i)
228  Ag_ang.col(i) += Ag_lin.col(i).cross(data.com[0]);
229 
230  data.hg = data.Ag*v;
231 
232  data.Ig.mass() = data.Ycrb[0].mass();
233  data.Ig.lever().setZero();
234  data.Ig.inertia() = data.Ycrb[0].inertia();
235 
236  return data.Ag;
237  }

285  template<typename JointModel>
286  static void algo(const se3::JointModelBase<JointModel> & jmodel,
287  se3::JointDataBase<typename JointModel::JointDataDerived> & jdata,
288  const se3::Model & model,
289  se3::Data & data)
290  {
291  typedef typename SizeDepType<JointModel::NV>::template ColsReturn<Data::Matrix6x>::Type ColsBlock;
292  typedef typename JointModel::JointDataDerived JointData;
293  typedef typename JointModel::Constraint_t Constraint_t;
294 
295 
296  const Model::JointIndex & i = (Model::JointIndex) jmodel.id();
297  const Model::Index & parent = model.parents[i];
298  const Motion & v = data.v[i];
299  const Inertia & Y = data.Ycrb[i];
300  const Inertia::Matrix6 & dY = data.dYcrb[i];
301 
302  data.Ycrb[parent] += data.liMi[i].act(Y);
303  data.dYcrb[parent] += SE3actOn(data.liMi[i],dY);
304 
305  // Calc Ag
306  ColsBlock Ag_cols = jmodel.jointCols(data.Ag);
307  jdata.U() = Y * jdata.S();
308  forceSet::se3Action(data.oMi[i],jdata.U(),Ag_cols);
309 
310  // Calc dAg = oXi* dU
311  typename JointData::U_t dU(dY * jdata.S()); // TODO: add dU inside jdata.
312  typename Constraint_t::DenseBase dS = jdata.S().variation(v);
313  dU.noalias() += Y.matrix()*dS;
314 
315  ColsBlock dAg_cols = jmodel.jointCols(data.dAg);
316  forceSet::se3Action(data.oMi[i],dU,dAg_cols);
317 
318  }

320  inline static Inertia::Matrix6 SE3actOn(const SE3 & M, const Inertia::Matrix6 & I)
321  {
322  typedef Inertia::Matrix6 Matrix6;
323  typedef SE3::Matrix3 Matrix3;
324  typedef SE3::Vector3 Vector3;
325  typedef Eigen::Block<const Matrix6,3,3> constBlock3;
326  typedef Eigen::Block<Matrix6,3,3> Block3;
327 
328  const constBlock3 & Ai = I.block<3,3> (Inertia::LINEAR, Inertia::LINEAR);
329  const constBlock3 & Bi = I.block<3,3> (Inertia::LINEAR, Inertia::ANGULAR);
330  const constBlock3 & Di = I.block<3,3> (Inertia::ANGULAR, Inertia::ANGULAR);
331 
332  const Matrix3 & R = M.rotation();
333  const Vector3 & t = M.translation();
334 
335  Matrix6 res;
336  Block3 Ao = res.block<3,3> (Inertia::LINEAR, Inertia::LINEAR);
337  Block3 Bo = res.block<3,3> (Inertia::LINEAR, Inertia::ANGULAR);
338  Block3 Co = res.block<3,3> (Inertia::ANGULAR, Inertia::LINEAR);
339  Block3 Do = res.block<3,3> (Inertia::ANGULAR, Inertia::ANGULAR);
340 
341  Ao = R*Ai*R.transpose();
342  Bo = R*Bi*R.transpose();
343  Do.row(0) = t.cross(Bo.col(0));
344  Do.row(1) = t.cross(Bo.col(1));
345  Do.row(2) = t.cross(Bo.col(2));
346 
347  Co.col(0) = t.cross(Ao.col(0));
348  Co.col(1) = t.cross(Ao.col(1));
349  Co.col(2) = t.cross(Ao.col(2));
350  Co += Bo.transpose();
351 
352  Bo = Co.transpose();
353  Do.col(0) += t.cross(Bo.col(0));
354  Do.col(1) += t.cross(Bo.col(1));
355  Do.col(2) += t.cross(Bo.col(2));
356  Do += R*Di*R.transpose();
357  return res;
358  }

362  inline const Data::Matrix6x &
363  dccrba(const Model & model, Data & data,
364  const Eigen::VectorXd & q,
365  const Eigen::VectorXd & v)
366  {
367  assert(model.check(data) && "data is not consistent with model.");
368  typedef Eigen::Block <Data::Matrix6x,3,-1> Block3x;
369 
370  forwardKinematics(model,data,q,v);
371  data.Ycrb[0].setZero();
372  for(Model::Index i=1;i<(Model::Index)(model.njoints);++i)
373  {
374  data.Ycrb[i] = model.inertias[i];
375  data.dYcrb[i] = model.inertias[i].variation(data.v[i]); // (vx* I - Ivx)
376  }
377 
378  for(Model::Index i=(Model::Index)(model.njoints-1);i>0;--i)
379  {
380  DCcrbaBackwardStep::run(model.joints[i],data.joints[i],
381  DCcrbaBackwardStep::ArgsType(model,data));
382  }
383  data.com[0] = data.Ycrb[0].lever();
384 
385  const Block3x Ag_lin = data.Ag.middleRows<3> (Force::LINEAR);
386  Block3x Ag_ang = data.Ag.middleRows<3> (Force::ANGULAR);
387  for (long i = 0; i<model.nv; ++i)
388  Ag_ang.col(i) += Ag_lin.col(i).cross(data.com[0]);
389 
390  data.hg = data.Ag*v;
391  data.vcom[0] = data.hg.linear()/data.Ycrb[0].mass();
392 
393  const Block3x dAg_lin = data.dAg.middleRows<3>(Force::LINEAR);
394  Block3x dAg_ang = data.dAg.middleRows<3>(Force::ANGULAR);
395  for (long i = 0; i<model.nv; ++i)
396  {
397  dAg_ang.col(i) += dAg_lin.col(i).cross(data.com[0]);// + Ag_lin.col(i).cross(data.vcom[0]);
398  }
399 
400  data.Ig.mass() = data.Ycrb[0].mass();
401  data.Ig.lever().setZero();
402  data.Ig.inertia() = data.Ycrb[0].inertia();
403 
404  return data.dAg;
405  }

421  inline bool CRBAChecker::checkModel_impl( const Model& model ) const
422  {
423  // For CRBA, the tree must be "compact", i.e. all descendants of a node i are stored
424  // immediately after i in the "parents" map, i.e. forall joint i, the interval i+1..n-1
425  // can be separated in two intervals [i+1..k] and [k+1..n-1], where any [i+1..k] is a descendant
426  // of i and none of [k+1..n-1] is a descendant of i.
427  for( JointIndex i=1; int(i)<model.njoints-1; ++i ) // no need to check joints 0 and N-1
428  {
429  JointIndex k=i+1;
430  while(internal::isDescendant(model,k,i)) ++k;
431  for( ; int(k)<model.njoints; ++k )
432  if( internal::isDescendant(model,k,i) ) return false;
433  }
434  return true;
435  }
