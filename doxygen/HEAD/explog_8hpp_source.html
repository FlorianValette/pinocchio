41  exp3(const Eigen::MatrixBase<D> & v)
42  {
43  EIGEN_STATIC_ASSERT_SAME_VECTOR_SIZE (Eigen::MatrixBase<D>,
44  Eigen::Vector3f);
45  assert (v.size () == 3);

48  return Eigen::AngleAxis<typename D::Scalar>(nv, v / nv).matrix();

61  template <typename D> Eigen::Matrix<typename D::Scalar,3,1,Eigen::internal::traits<D>::Options>
62  log3(const Eigen::MatrixBase<D> & R)
63  {
64  EIGEN_STATIC_ASSERT_SAME_MATRIX_SIZE(D, Eigen::Matrix3d);
65  Eigen::AngleAxis<typename D::Scalar> angleAxis(R);
66  assert(0 <= angleAxis.angle() && angleAxis.angle() <= 2 * M_PI);
67  if (angleAxis.angle() > M_PI)
68  return -(2*M_PI - angleAxis.angle()) * angleAxis.axis();
69  return angleAxis.axis() * angleAxis.angle();

80  template <typename _Scalar, int _Options> SE3Tpl<_Scalar, _Options>
81  exp6(const MotionTpl<_Scalar,_Options> & nu)
82  {
83  typedef _Scalar Scalar;
84  typedef typename MotionTpl<Scalar,_Options>::Vector3 Vector3;
85  typedef typename MotionTpl<Scalar,_Options>::Matrix3 Matrix3;
86 
87  const Vector3 & w = nu.angular();
88  const Vector3 & v = nu.linear();
89  Scalar t = w.norm();
90  if (t > 1e-15)
91  {
92  const double inv_t = 1./t;
93  Matrix3 S(alphaSkew(inv_t, w));
94  double ct,st; SINCOS (t,&st,&ct);
95  Matrix3 V((1. - ct) * inv_t * S + (1. - st * inv_t) * S * S);
96  Vector3 p(v + V * v);
97  return SE3Tpl<_Scalar, _Options>(exp3(w), p);
98  }
99  else
100  {
101  return SE3Tpl<_Scalar, _Options>(Matrix3::Identity(), v);
102  }
103  }

130  MotionTpl<_Scalar,_Options>
131  log6(const SE3Tpl<_Scalar, _Options> & M)
132  {
133  typedef _Scalar Scalar;
134  typedef typename SE3Tpl<Scalar,_Options>::Vector3 Vector3;
135  typedef typename SE3Tpl<Scalar,_Options>::Matrix3 Matrix3;
136 
137  const Matrix3 & R = M.rotation();
138  const Vector3 & p = M.translation();
139  Vector3 w(log3(R));
140  Scalar t = w.norm();
141  if (t > 1e-15)
142  {
143  Matrix3 S(alphaSkew(1./t, w));
144  double ct,st; SINCOS (t,&st,&ct);
145  Matrix3 V(Matrix3::Identity() + (1. - ct)/t * S + (1. - st/t) * S * S);
146 
147  return MotionTpl<_Scalar,_Options>(V.inverse() * p, w);
148  }
149  else
150  return MotionTpl<_Scalar,_Options>(p, w);
151 
152  }
