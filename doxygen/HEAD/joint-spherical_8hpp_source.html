321  void calc_aba(JointDataDerived & data, Inertia::Matrix6 & I, const bool update_I) const
322  {
323  data.U = I.block<6,3> (0,Inertia::ANGULAR);
324  data.Dinv = I.block<3,3> (Inertia::ANGULAR,Inertia::ANGULAR).inverse();
325  data.UDinv.middleRows<3> (Inertia::ANGULAR).setIdentity(); // can be put in data constructor
326  data.UDinv.middleRows<3> (Inertia::LINEAR) = data.U.block<3,3> (Inertia::LINEAR, 0) * data.Dinv;
327 
328  if (update_I)
329  {
330  I.block<6,3> (0,Inertia::ANGULAR).setZero();
331  I.block<3,3> (Inertia::ANGULAR,Inertia::LINEAR).setZero();
332  I.block<3,3> (Inertia::LINEAR,Inertia::LINEAR) -= data.UDinv.middleRows<3> (Inertia::LINEAR) * I.block<3,3> (Inertia::ANGULAR, Inertia::LINEAR);

342  ConfigVector_t integrate_impl(const Eigen::VectorXd & qs,const Eigen::VectorXd & vs) const
